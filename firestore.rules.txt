rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isSignedIn() { return request.auth != null; }
    match /rooms/{roomId} {
      allow read, write: if isSignedIn();
      match /members/{uid} { allow read, write: if isSignedIn(); }
      match /alarms/{alarmId} { allow read, write: if isSignedIn(); }
      match /events/{eventId} { allow read, write: if isSignedIn(); }
    }
  }
}
